Решение:
from turtle import *

screensize(200, 200)
shape("turtle")
k = 20
pendown()
tracer(10)
for _ in range(10):
    forward(10 * k)
    right(60)
penup()
for x in range(-25, 25):
    for y in range(-20, 30):
        goto(x*k, y*k)
        dot(3)
done()
